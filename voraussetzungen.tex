\begin{frame}{Freie Software}
	4 Freiheiten:
	\begin{itemize}
		\item Verwenden\only<handout>{ - Die Freiheit, die Software uneingeschränkt und für jeden Zweck einzusetzen.}
		\item Verstehen\only<handout>{ - Die Freiheit, die Funktionsweise der Software untersuchen und verstehen zu können.}
		\item Verbreiten\only<handout>{ - Die Freiheit, Kopien der Software zu verbreiten, um damit seinen Mitmenschen zu helfen.}
		\item Verbessern\only<handout>{ - Die Freiheit, die Software zu verbessern und die Verbesserungen an die Öffentlichkeit weiterzugeben, sodass die gesamte Gesellschaft davon profitieren kann.}
	\end{itemize}
	\url{http://www.inf-schule.de/software/freie\_software/02\_freiheiten\_freier\_software}
\end{frame}

\begin{frame}{FLOSS}
	\begin{description}
		\item[Freie Software] Anwender, Gesellschaft, ethisch, sozial
		\item[Open Source] Entwickler, Business, praktisch, pragmatisch
		\item[FLOSS] Free/Libre Open Source Software
	\end{description}
	\url{https://de.wikipedia.org/wiki/Freie\_Software\#Open\_Source}
\end{frame}
